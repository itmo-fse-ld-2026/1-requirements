\makeatletter
\newcommand{\reqlabel}[2][]{%
  % #1 = optional label (e.g., auth)
  % #2 = prefix (e.g., REQ)
  \ifcsname c@#2\endcsname\else\newcounter{#2}\fi
  \stepcounter{#2}%
  \edef\myid{#2-\ifnum\value{#2}<10 0\fi\arabic{#2}}%
  \myid
  \ifx&#1&\else
    \def\@currentlabel{\myid}%
    \label{#1}%
  \fi
}

\newcommand{\reqref}[2][]{%
  % #1 = optional description
  % #2 = label name
  \ref{#2}\ifx&#1&\else\ (#1)\fi
}

\newcommand{\resetreqcounter}[1]{%
  \ifcsname c@#1\endcsname
    \setcounter{#1}{0}%
  \else
    \newcounter{#1}\setcounter{#1}{0}%
  \fi
}
\makeatother
