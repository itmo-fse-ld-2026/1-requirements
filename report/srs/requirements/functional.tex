\subsubsection{Functional Requirements}

\resetreqcounter{GeFR}
\begin{reqtable}[
  caption = {Общие функциональные требования к Сайту.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel{GeFR} & Шапка Сайта \textit{Должна} содержать следующие \mbox{элементы}:
  \begin{compactitem}
    \item логотип Сайта;
    \item список доступных гаданий;
    \item поиск по содержимому страниц Сайта.
  \end{compactitem} \\
  \reqlabel{GeFR} & Сайт \textit{Должен} показывать Пользователям рекламу следующих форматов:
  \begin{compactitem}
    \item \textit{баннерная:} между блоками Сайта;
    \item \textit{всплывающая:} в углу экрана.
  \end{compactitem} \\
  \reqlabel{GeFR} & Сайт \textit{Должен} вести статистику посещений Пользователей. \\
\end{reqtable}

\resetreqcounter{MpFR}
\begin{reqtable}[
  caption = {Функциональные требования к главной странице Сайта.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel{MpFR} & Главная страница Сайта \textit{Должна} содержать блок с новыми способами гадания. \\
  \reqlabel{MpFR} & Главная страница Сайта \textit{Должна} содержать раздел со способами гадания, которые актуальны на текущую дату; например:
  \begin{compactitem}
    \item на 18 марта;
    \item на весну;
    \item на четверг.
  \end{compactitem} \\
\end{reqtable}

\resetreqcounter{LgFR}
\begin{reqtable}[
  caption = {Юридические функциональные требования к Сайту.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel{LgFR} & Сайт \textit{Должен} содержать страницу с Пользовательским соглашением. \\
  \reqlabel{LgFR} & Сайт \textit{Должен} содержать страницу с Политикой конфиденциальности. \\
  \reqlabel{LgFR} & Сайт \textit{Должен} содержать страницу со своим описанием для Пользователя. \\
  \reqlabel{LgFR} & Сайт \textit{Должен} содержать страницу с контактами Администраторов в случае возникновения вопросов. \\
\end{reqtable}

\resetreqcounter{DvFR}
\begin{reqtable}[
  caption = {Функциональные требования к процессам гадания на Сайте.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel{DvFR} & Гадания на Сайте \textit{Должны} быть интерактивными, то есть взаимодействовать с Пользователем для получения результата. \\
  \reqlabel{DvFR} & Сайт \textit{Должен} предоставить толкование результата, полученного в процессе гадания. \\
  \reqlabel{DvFR} & Гадания на Сайте \textit{Должны} быть сгруппированы следующим образом:
  \begin{compactitem}
    \item \textit{по тематике:} бытовые, любовные и пр.;
    \item \textit{по разделу:} оракул, карты Таро, руны и пр.
  \end{compactitem} \\
  \reqlabel{DvFR} & Группа \textit{Должна} предоставить следующие критерии сортировки:
  \begin{compactitem}
    \item по дате появления гадания на Сайте;
    \item по популярности гадания на Сайте;
    \item по рейтингу гадания на Сайте.
  \end{compactitem} \\
  \reqlabel{DvFR} & Гадания на Сайте \textit{Должны} содержать блоки со своим описанием, которое включает следующее:
  \begin{compactitem}
    \item название услуги;
    \item цель услуги;
    \item формат предоставления услуги;
    \item результат оказания услуги.
  \end{compactitem} \\
\end{reqtable}

\resetreqcounter{FbFR}
\begin{reqtable}[
  caption = {Функциональные требования к блоку обратной связи.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel{FbFR} & Страницы Сайта \textit{Должны} быть оснащены блоком обратной связи, который доступен Пользователям. \\
  \reqlabel{FbFR} & Блок обратной связи \textit{Должен} содержать следующие элементы:
  \begin{compactitem}
    \item рейтинг текущей страницы;
    \item форма отправки комментария;
    \item список опубликованных комментариев;
    \item кнопки для публикации в социальных сетях ссылки на страницу;
  \end{compactitem} \\
  \reqlabel{FbFR} & Рейтинг \textit{Должен} выражаться в десятибалльной шкале, где $1$ --- неудовлетворительно, $10$ --- замечательно. \\
  \reqlabel{FbFR} & Пользователи \textit{Должны} иметь возможность влиять на рейтинг любой доступной страницы Сайта. \\
  \reqlabel{FbFR} & Пользователи \textit{Должны} иметь возможность оставлять комментарии при помощи формы. \\
  \reqlabel{FbFR} & Форма отправки комментария \textit{Должна} содержать следующие поля:
  \begin{compactitem}
    \item псевдоним автора комментария;
    \item содержание комментария.
  \end{compactitem} \\
  \reqlabel{FbFR} & Поля формы отправки комментария \textit{Должны} быть обязательными к заполнению и проверяться на валидность, то есть быть непустыми. \\
  \reqlabel{FbFR} & Опубликованные комментарии в списке \textit{Должны} должны содержать следующую информацию:
  \begin{compactitem}
    \item дата публикации комментария;
    \item псевдоним автора комментария;
    \item содержание комментария.
  \end{compactitem} \\
  \reqlabel{FbFR} & Комментарии \textit{Должны} быть отсортированы по убыванию даты публикации, то есть от новых к старым. \\
  \reqlabel{FbFR} & Список опубликованных комментариев \textit{Должен} отображаться Пользователю с применением Пагинации. \\
\end{reqtable}

\resetreqcounter{FvFR}
\begin{reqtable}[
  caption = {Функциональные требования к модулю избранных страниц.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel{FvFR} & Страницы Сайта \textit{Должны} быть оснащены панелью избранных странц. \\
  \reqlabel{FvFR} & Панель избранных страниц \textit{Должна} содержать следующие элементы:
  \begin{compactitem}
    \item кнопка добавления текущей страницы в избранное;
    \item ссылка на избранные страницы.
  \end{compactitem} \\
  \reqlabel{FvFR} & Страница «Избранное» \textit{Должна} содержать индивидуальный список избранных страниц Пользователя. \\
  \reqlabel{FvFR} & Элемент списка избранных страниц \textit{Должен} обладать следующими компонентами:
  \begin{compactitem}
    \item название страницы;
    \item описание страницы;
    \item иконка страницы;
    \item кнопка удаления страницы из списка;
    \item кнопка просмотра страницы.
  \end{compactitem} \\
\end{reqtable}
