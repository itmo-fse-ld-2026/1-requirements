\resetreqcounter{GeFR}
\begin{srstable}[
  caption = {Общие функциональные требования к Системе.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:gef1]{GeFR} & Система \textit{Должна} обеспечивать интеграцию в заголовочную область интерфейса (Шапку) следующих функциональных компонентов:
  \begin{compactitem}
    \item графический идентификатор (логотип) Системы, обеспечивающий навигационную адресацию на главную страницу;
    \item структурированный перечень доступных методик прогнозирования, инициирующий перенаправление на соответствующие целевые страницы;
    \item модуль полнотекстового поиска по контенту страниц Системы.
  \end{compactitem} \\
  \reqlabel[ref:gef2]{GeFR} & Система \textit{Должна} осуществлять демонстрацию рекламно-информационных материалов в следующих регламентированных форматах:
  \begin{compactitem}
    \item \textit{баннерный:} с позиционированием между информационными блоками Системы;
    \item \textit{всплывающий:} с позиционированием в угловой области экранного пространства.
  \end{compactitem} \\
  \reqlabel[ref:gef3]{GeFR} & Система \textit{Должна} осуществлять непрерывный автоматизированный сбор и агрегацию статистических показателей активности Пользователей. \\
\end{srstable}

\resetreqcounter{DvFR}
\begin{srstable}[
  caption = {Функциональные требования к процессам гадания в Системе.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:dvf1]{DvFR} & Система \textit{Должна} поддерживать реализацию процедур гадания, предусматривающих применение одного из следующих методов интерактивного взаимодействия с Пользователем для генерации результата:
  \begin{compactitem}
    \item ввод текстового запроса в неструктурированном виде;
    \item выбор одного или множества вариантов из фиксированного перечня;
    \item инициация действия посредством активации управляющего элемента без требования ввода данных;
    \item выбор целевого объекта из представленного визуального ряда.
  \end{compactitem} \\
  \reqlabel[ref:dvf2]{DvFR} & Система \textit{Должна} обеспечивать генерацию и вывод интерпретации (толкования) результата, полученного по завершении процесса гадания. \\
  \reqlabel[ref:dvf3]{DvFR} & Система \textit{Должна} осуществлять классификацию методик гадания по следующим признакам:
  \begin{compactitem}
    \item \textit{по тематической направленности:} бытовые, любовные и аналогичные;
    \item \textit{по категориальной принадлежности:} оракул, карты Таро, руны и аналогичные.
  \end{compactitem} \\
  \reqlabel[ref:dvf4]{DvFR} & Система \textit{Должна} предоставлять функционал ранжирования элементов внутри групп по следующим параметрам:
  \begin{compactitem}
    \item дата интеграции методики в Систему;
    \item индекс популярности методики в рамках Системы;
    \item агрегированный пользовательский рейтинг методики.
  \end{compactitem} \\
  \reqlabel[ref:dvf5]{DvFR} & Система \textit{Должна} сопровождать каждую методику гадания информационно-справочными блоками, включающими:
  \begin{compactitem}
    \item наименование предоставляемой услуги;
    \item целевое назначение услуги;
    \item регламентированный формат предоставления услуги;
    \item характеристику ожидаемого результата оказания услуги.
  \end{compactitem} \\
\end{srstable}

\resetreqcounter{FbFR}
\begin{srstable}[
  caption = {Функциональные требования к блоку обратной связи.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:fbf1]{FbFR} & Система \textit{Должна} обеспечивать интеграцию модуля обратной связи, открытого для взаимодействия со стороны Пользователей, на всех подлежащих оценке страницах. \\
  \reqlabel[ref:fbf2]{FbFR} & Система \textit{Должна} обеспечивать функционирование модуля обратной связи, состоящего из следующих структурных компонентов:
  \begin{compactitem}
    \item индикатор текущего рейтингового показателя страницы;
    \item интерактивная форма для отправки пользовательского комментария;
    \item реестр ранее опубликованных комментариев;
    \item элементы управления для дистрибуции гиперссылки на страницу в сторонних социальных сетях.
  \end{compactitem} \\
  \reqlabel[ref:fbf3]{FbFR} & Система \textit{Должна} обеспечивать расчет и отображение рейтинга на основе десятибалльной шкалы оценивания (где $1$ соответствует значению «неудовлетворительно», а $10$ — «замечательно»). \\
  \reqlabel[ref:fbf4]{FbFR} & Система \textit{Должна} предоставлять Пользователям функциональную возможность оказывать влияние на совокупный показатель рейтинга любой доступной страницы Системы. \\
  \reqlabel[ref:fbf5]{FbFR} & Система \textit{Должна} предоставлять Пользователям техническую возможность публикации сообщений посредством заполнения специализированной формы. \\
  \reqlabel[ref:fbf6]{FbFR} & Система \textit{Должна} обеспечивать наличие в модуле комментариев следующих полей:
  \begin{compactitem}
    \item псевдоним (идентификатор) автора;
    \item содержательная часть сообщения.
  \end{compactitem} \\
  \reqlabel[ref:fbf7]{FbFR} & Система \textit{Должна} осуществлять автоматизированную процедуру валидации полей формы на предмет обязательного заполнения (отсутствия пустых значений) до момента инициации отправки. \\
  \reqlabel[ref:fbf8]{FbFR} & Система \textit{Должна} отображать реестр опубликованных комментариев с указанием следующих метаданных:
  \begin{compactitem}
    \item хронологическая метка публикации;
    \item псевдоним автора;
    \item содержательная часть сообщения.
  \end{compactitem} \\
  \reqlabel[ref:fbf9]{FbFR} & Система \textit{Должна} осуществлять ранжирование опубликованных комментариев в порядке обратной хронологии (от более поздних к более ранним). \\
  \reqlabel[ref:fbf10]{FbFR} & Система \textit{Должна} реализовывать алгоритм постраничного вывода (Пагинации) при отображении списка комментариев. \\
\end{srstable}
