\subsubsection{Non-Functional Requirements}

\resetreqcounter{PfNFR}
\begin{srstable}[
  caption = {Нефункциональные требования к производительности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:pfnf1]{PfNFR} & Сайт \textit{Должен} обеспечивать время загрузки для любой страницы не более 2 секунд для 95\% запросов при нормальной нагрузке (до 500 одновременных сессий). \\
  \reqlabel[ref:pfnf2]{PfNFR} & Сайт \textit{Должен} обеспечивать время ответа сервера при генерации предсказания не более 3 секунды. \\
  \reqlabel[ref:pfnf3]{PfNFR} & Сайт \textit{Должен} обеспечивать время ответа сервера при асинхронном обновлении страницы не более 1 секунды. \\
  \reqlabel[ref:pfnf4]{PfNFR} & Сайт \textit{Должен} обрабатывать до 500 одновременных активных сессий без потери производительности. \\
  \reqlabel[ref:pfnf5]{PfNFR} & Сайт \textit{Должен} поддерживать суточную аудиторию до 30000 уникальных пользователей. \\
  \reqlabel[ref:pfnf6]{PfNFR} & Сайт \textit{Должен} сохранять работоспособность при пиковых нагрузках до 200 запросов в секунду. \\
\end{srstable}

\resetreqcounter{AvNFR}
\begin{srstable}[
  caption = {Нефункциональные требования к доступности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:avnf1]{AvNFR} & Сайт \textit{Должен} обеспечивать доступность не менее 99.5\% времени в месяц, за исключением плановых технических работ. \\
  \reqlabel[ref:avnf2]{AvNFR} & Сайт \textit{Должен} быть недоступен Пользователям при проведении плановых технических работ, предварительно согласованных с заказчиком, в периоды наименьшей активности Пользовтелей (02:00-06:00 по московскому времени). \\
  \reqlabel[ref:avnf3]{AvNFR} & Сайт \textit{Должен} обеспечивать время восстановления после аварийного отказа (RTO) не более 4 часов. \\
\end{srstable}

\resetreqcounter{RlNFR}
\begin{srstable}[
  caption = {Нефункциональные требования к надежности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:rlnf1]{RlNFR} & Сайт \textit{Должен} корректно обрабатывать ошибки ввода данных (некорректный формат, пустые поля и др.) без сбоев и потери данных. \\
  \reqlabel[ref:rlnf2]{RlNFR} & Сайт \textit{Должен} выполнять автоматическое резервное копирование базы данных не реже 1 раза в сутки. \\
  \reqlabel[ref:rlnf3]{RlNFR} & Сайт \textit{Должен} обеспечивать максимально допустимую потерю данных при аварийном отказе (RPO) не более 24 часов. \\
  \reqlabel[ref:rlnf4]{RlNFR} & Сайт \textit{Должен} сохранять работоспособность при отказе внешних зависимостей. \\
\end{srstable}

\resetreqcounter{ScNFR}
\begin{srstable}[
  caption = {Нефункциональные требования к безопасности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:scnf1]{ScNFR} & Сайт \textit{Должен} осуществлять весь обмен данными между клиентом и сервером по защищенному протоколу HTTPS. \\
  \reqlabel[ref:scnf2]{ScNFR} & Сайт \textit{Должен} предоставлять доступ к Панели управления только после успешной аутентификации Администратора. \\
  \reqlabel[ref:scnf3]{ScNFR} & Сайт \textit{Должен} хранить секреты в базе данных в хэшированном виде с использованием Соли. \\
  \reqlabel[ref:scnf4]{ScNFR} & Сайт \textit{Должен} быть защищена от следующих уязвимостей:
  \begin{compactitem}
    \item XSS-атаки;
    \item SQL-инъекции;
    \item DDoS-атаки.
  \end{compactitem} \\
  \reqlabel[ref:scnf5]{ScNFR} & Сайт \textit{Должен} применять к IP-адресу временную блокировку на 15 минут после 5 неудачных попыток входа в панель управления. \\
\end{srstable}

\begin{srstable}[
  caption = {Нефункциональные требования к использованию.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:usnf1]{UsNFR} & Сайт \textit{Должен} предоставлять интерфейс с поддержкой адаптивную верстку с переключением между мобильным (ширина экрана менее 768 пикселей) и десктопным (768 пикселей и более) режимами отображения.   \reqlabel[ref:usnf3]{UsNFR} & Сайт \textit{Должен} предоставлять понятные Пользователю сообщения об ошибках валидации для всех форм. \\
\end{srstable}
