\subsubsection{Non-Functional Requirements}

\resetreqcounter{PfNFR}
\begin{srstable}[
  caption = {Требования к производительности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:pfnf1]{PfNFR} & Система \textit{Должна} обеспечивать время загрузки любой страницы не более 2 (двух) секунд для 95\% запросов при нормальной эксплуатационной нагрузке, определяемой как наличие до 500 одновременных сессий. \\
  \reqlabel[ref:pfnf2]{PfNFR} & Система \textit{Должна} обеспечивать время ответа сервера при генерации предсказания не более 3 (трёх) секунд. \\
  \reqlabel[ref:pfnf3]{PfNFR} & Система \textit{Должна} обеспечивать время ответа сервера при асинхронном обновлении содержимого страницы не более 1 (одной) секунды. \\
  \reqlabel[ref:pfnf4]{PfNFR} & Система \textit{Должна} быть способна обрабатывать до 500 одновременных активных сессий без возникновения деградации производительности. \\
  \reqlabel[ref:pfnf5]{PfNFR} & Система \textit{Должна} быть способна поддерживать суточную аудиторию до 30\,000 уникальных пользователей. \\
  \reqlabel[ref:pfnf6]{PfNFR} & Система \textit{Должна} сохранять работоспособность при пиковой нагрузке до 200 запросов в секунду. \\
\end{srstable}

\resetreqcounter{AvNFR}
\begin{srstable}[
  caption = {Требования к доступности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:avnf1]{AvNFR} & Система \textit{Должна} обеспечивать уровень доступности не менее 99.5\% времени в месяц, за исключением периодов проведения плановых технических работ. \\
  \reqlabel[ref:avnf2]{AvNFR} & Система \textit{Должна} осуществлять плановые технические работы в периоды наименьшей активности пользователей (02:00–06:00 по московскому времени) при условии предварительного письменного согласования с заказчиком. В указанные периоды доступность Системы не гарантируется. \\
  \reqlabel[ref:avnf3]{AvNFR} & Система \textit{Должна} обеспечивать время восстановления после аварийного отказа (Recovery Time Objective, RTO) не более 4 (четырёх) часов. \\
\end{srstable}

\resetreqcounter{RlNFR}
\begin{srstable}[
  caption = {Требования к надёжности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:rlnf1]{RlNFR} & Система \textit{Должна} корректно обрабатывать ошибки ввода данных (включая, но не ограничиваясь: некорректный формат, пустые поля) без возникновения сбоев и потери данных. \\
  \reqlabel[ref:rlnf2]{RlNFR} & Система \textit{Должна} осуществлять автоматическое резервное копирование базы данных с периодичностью не реже одного раза в сутки. \\
  \reqlabel[ref:rlnf3]{RlNFR} & Система \textit{Должна} обеспечивать максимально допустимую потерю данных при аварийном отказе (Recovery Point Objective, RPO) не более 24 (двадцати четырёх) часов. \\
  \reqlabel[ref:rlnf4]{RlNFR} & Система \textit{Должна} сохранять полную или частичную работоспособность при отказе внешних зависимостей (включая, но не ограничиваясь: API социальных сетей, поисковые системы). В случае отказа внешних зависимостей базовая функциональность Системы (просмотр гаданий, получение предсказаний) должна оставаться доступной. \\
\end{srstable}

\resetreqcounter{ScNFR}
\begin{srstable}[
  caption = {Требования к безопасности.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:scnf1]{ScNFR} & Система \textit{Должна} осуществлять весь обмен данными между клиентской частью и серверной частью с использованием защищённого протокола HTTPS. \\
  \reqlabel[ref:scnf2]{ScNFR} & Система \textit{Должна} предоставлять доступ к Панели управления исключительно после успешного прохождения процедуры аутентификации Администратора. \\
  \reqlabel[ref:scnf3]{ScNFR} & Система \textit{Должна} осуществлять хранение секретов (включая, но не ограничиваясь: пароль Администратора) в базе данных в хэшированном виде с использованием соли. \\
  \reqlabel[ref:scnf4]{ScNFR} & Система \textit{Должна} быть защищена от следующих типов уязвимостей:
  \begin{compactitem}
    \item межсайтовый скриптинг (Cross-Site Scripting, XSS);
    \item внедрение SQL-кода (SQL-инъекции);
    \item распределённые атаки типа «отказ в обслуживании» (DDoS-атаки).
  \end{compactitem} \\
  \reqlabel[ref:scnf5]{ScNFR} & Система \textit{Должна} применять к IP-адресу временную блокировку продолжительностью 15 (пятнадцать) минут после 5 (пяти) неудачных попыток аутентификации в Панели управления. \\
\end{srstable}

\resetreqcounter{UsNFR}
\begin{srstable}[
  caption = {Требования к удобству использования.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:usnf1]{UsNFR} & Система \textit{Должна} предоставлять пользовательский интерфейс, реализованный с применением адаптивной вёрстки, обеспечивающей переключение между мобильным режимом отображения (при ширине экрана менее 768 пикселей) и десктопным режимом отображения (при ширине экрана 768 пикселей и более). При ширине экрана свыше 1920 пикселей содержимое интерфейса подлежит центрированию с фиксацией максимальной ширины для сохранения читаемости. \\
  \reqlabel[ref:usnf2]{UsNFR} & Система \textit{Должна} предоставлять явные визуальные признаки (изменение цвета, подчёркивание, анимация) при наведении курсора на все интерактивные элементы (кнопки, ссылки, поля ввода). \\
  \reqlabel[ref:usnf3]{UsNFR} & Система \textit{Должна} предоставлять понятные Пользователю сообщения об ошибках валидации для всех форм ввода данных, с указанием конкретного поля и способа устранения ошибки. \\
\end{srstable}

\resetreqcounter{PtNFR}
\begin{srstable}[
  caption = {Требования к переносимости.},
]{colspec = {Q[c, m] X[j, m]}}
  Идентификатор & Описание \\
  \reqlabel[ref:ptnf1]{PtNFR} & Система \textit{Должна} корректно функционировать во всех современных браузерах, включая, но не ограничиваясь:
  \begin{compactitem}
    \item Google Chrome;
    \item Mozilla Firefox;
    \item Microsoft Edge;
    \item Safari;
    \item Яндекс.Браузер.
  \end{compactitem} \\
  \reqlabel[ref:ptnf2]{PtNFR} & Система \textit{Должна} корректно функционировать без необходимости установки дополнительных программных модулей (плагинов) или расширений браузера. \\
\end{srstable}
